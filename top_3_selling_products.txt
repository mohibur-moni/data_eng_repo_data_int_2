--calculating top three selling products--

-- Based on quantity--
SELECT ProductName , UnitsOnOrder, UnitPrice * UnitsOnOrder 
from Products p
order by 2 desc, 1 
limit 3;

-- based on price per-unit
SELECT ProductName , UnitsOnOrder , UnitPrice * UnitsOnOrder 
from Products p
order by 3 desc, 1 
limit 3;
